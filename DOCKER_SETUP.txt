================================================================================
COMMON DOCKER COMPOSE COMMANDS
================================================================================

Start:
  docker-compose up -d                     # Start services
  docker-compose up -d --build             # Build and start

Stop/Remove:
  docker-compose down                      # Stop (keep images)
  docker-compose down -v --rmi all         # Stop and remove everything

Logs:
  docker-compose logs -f                   # Follow all logs
  docker-compose logs -f backend           # Backend logs only
  docker-compose logs --since 10m          # Last 10 minutes

Status:
  docker-compose ps                        # Show status
  docker-compose config --quiet            # Validate config

Build:
  docker-compose build                     # Build images
  docker-compose build --no-cache          # Force rebuild
  docker-compose build --pull              # Fresh base images

Execute:
  docker-compose exec backend python -m pytest     # Run tests
  docker-compose exec backend sh                   # Shell access
  docker-compose exec frontend npm test            # Frontend tests

================================================================================
MAKEFILE COMMANDS
================================================================================

Setup:
  make init               # Create .env from .env.example
  make build              # Build Docker images
  make build-no-cache     # Force rebuild without cache

Running:
  make up                 # Start services (production)
  make dev                # Start services (development)
  make down               # Stop all services
  make restart            # Restart all services
  make ps                 # Show service status

Logs:
  make logs               # Follow all logs
  make logs-backend       # Backend logs only
  make logs-frontend      # Frontend logs only

Testing:
  make test               # Run all tests
  make test-backend       # Backend tests only
  make test-frontend      # Frontend tests only

Shell Access:
  make shell-backend      # Backend shell
  make shell-frontend     # Frontend shell

Cleanup:
  make clean              # Stop and remove containers (keep images)
  make clean-all          # Remove everything
  make prune              # Remove all unused Docker resources

Advanced:
  make config             # Show Docker Compose config
  make validate           # Validate Docker Compose file
  make health             # Check service health
  make info               # Show Docker info

Run `make help` for all commands.
